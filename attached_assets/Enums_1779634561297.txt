public enum UserType
{
    /// <summary>
    /// مواطن - الفرد العادي الذي يستخدم خدمات التحقق من الهوية
    /// له الحق في رؤية بياناته وإدارة موافقاته فقط
    /// </summary>
    Citizen = 1,

    /// <summary>
    /// موظف مؤسسة - موظف يعمل في إحدى المؤسسات المسجلة
    /// له صلاحيات محدودة لطلب التحقق من بيانات المواطنين
    /// لا يستطيع الوصول إلى بيانات حساسة أخرى
    /// </summary>
    InstitutionStaff = 2,

    /// <summary>
    /// إدارة مؤسسة - مسؤول إداري في المؤسسة
    /// له صلاحيات أعلى من الموظف العادي
    /// يستطيع إدارة المستخدمين والإعدادات الخاصة بالمؤسسة
    /// </summary>
    InstitutionAdmin = 3,

    /// <summary>
    /// مسؤول النظام (Super Admin) - إدارة النظام الكاملة
    /// له أعلى الصلاحيات في النظام
    /// يستطيع إدارة جميع المؤسسات والمستخدمين والإعدادات
    /// يمكنه الوصول إلى جميع السجلات والبيانات
    /// </summary>
    SuperAdmin = 4,

    /// <summary>
    /// موظف مصلحة الأحوال المدنية - موظف حكومي في الأحوال المدنية
    /// له صلاحيات خاصة للتحقق من البيانات الأساسية
    /// يستطيع استيراد وتحديث بيانات المواطنين
    /// يمكنه الوصول إلى كل بيانات المواطنين لأغراض إدارية
    /// </summary>
    CivilRegistry = 5
}



public enum AuditAction
{
    /// <summary>
    /// تسجيل دخول ناجح للنظام
    /// </summary>
    Login = 1,

    /// <summary>
    /// محاولة تسجيل دخول فاشلة (كلمة مرور خاطئة أو مستخدم غير موجود)
    /// </summary>
    LoginFailed = 2,

    /// <summary>
    /// التحقق الناجح من الوجه (Face Verification)
    /// يسجل عندما يتم التحقق من هوية المستخدم عبر التعرف على الوجه
    /// </summary>
    FaceVerified = 3,

    /// <summary>
    /// فشل التحقق من الوجه
    /// يحدث عندما لا يتطابق الوجه المجمع مع الوجه المخزن
    /// </summary>
    FaceFailed = 4,

    /// <summary>
    /// إرسال رمز التحقق لمرة واحدة (OTP) بنجاح
    /// يسجل عند إرسال OTP إلى البريد الإلكتروني أو الهاتف
    /// </summary>
    OtpSent = 5,

    /// <summary>
    /// التحقق الناجح من رمز OTP
    /// يسجل عندما يدخل المستخدم الرمز الصحيح
    /// </summary>
    OtpVerified = 6,

    /// <summary>
    /// فشل التحقق من رمز OTP
    /// يحدث عندما يدخل المستخدم رمز خاطئ أو منتهي الصلاحية
    /// </summary>
    OtpFailed = 7,

    /// <summary>
    /// التحقق الناجح من الهوية
    /// يسجل النجاح النهائي في عملية التحقق من هوية المواطن
    /// </summary>
    IdentityVerified = 8,

    /// <summary>
    /// إصدار وثيقة جديدة للمواطن
    /// يسجل عند إصدار أي وثيقة رسمية (شهادة، بطاقة، إلخ)
    /// </summary>
    DocumentIssued = 9,

    /// <summary>
    /// التحقق من صحة وثيقة موجودة
    /// يسجل عند تحقق المؤسسة من صحة وثيقة ما
    /// </summary>
    DocumentVerified = 10,

    /// <summary>
    /// تعليق حساب المواطن مؤقتاً
    /// يمنع المواطن من استخدام الخدمة لكن البيانات تبقى محفوظة
    /// </summary>
    AccountSuspended = 11,

    /// <summary>
    /// تفعيل حساب موقوف سابقاً
    /// يسمح للمواطن باستخدام الخدمة مجدداً
    /// </summary>
    AccountActivated = 12,

    /// <summary>
    /// تسجيل بيانات الوجه البيومترية للمستخدم
    /// يسجل عند حفظ قالب الوجه الأول للمستخدم
    /// </summary>
    FaceRegistered = 13,

    /// <summary>
    /// استيراد جماعي لبيانات المواطنين
    /// يسجل عند استيراد آلاف المواطنين من قاعدة الأحوال المدنية
    /// </summary>
    BulkImport = 14
}


public enum CitizenStatus
{
    /// <summary>
    /// انتظار التفعيل - تم استيراد بيانات المواطن لكن لم يفعل حسابه بعد
    /// المواطن لا يستطيع استخدام الخدمات حتى يتم التفعيل
    /// </summary>
    PendingActivation = 1,

    /// <summary>
    /// مفعل وجاهز - المواطن قد فعل حسابه ويمكنه استخدام جميع الخدمات بشكل طبيعي
    /// هذه هي الحالة الطبيعية للمواطن النشط
    /// </summary>
    Active = 2,

    /// <summary>
    /// موقوف مؤقتاً - تم تعليق حساب المواطن مؤقتاً (قد يكون بسبب نشاط مريب أو انتهاك)
    /// البيانات محفوظة لكن المواطن لا يستطيع استخدام الخدمات حالياً
    /// يمكن إعادة التفعيل لاحقاً إذا تم حل المشكلة
    /// </summary>
    Suspended = 3,

    /// <summary>
    /// ملغى نهائياً - تم إلغاء حساب المواطن بشكل دائم ولا يمكن استرجاعه
    /// قد يكون بسبب وفاة المواطن أو انتهاك خطير أو قرار إداري نهائي
    /// يظل السجل محفوظاً للأغراض التاريخية والقانونية
    /// </summary>
    Revoked = 4
}
